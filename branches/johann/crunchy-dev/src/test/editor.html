def mandel(x, y, maxiter = 20):
    c = complex(x, y)
    z = complex(0, 0)
    for k in range(maxiter):
        z = z*z + c
    if abs(z) < 2.0:
        return True
    else:
        return False

print mandel(-0.2, 0.1)
print mandel(1.0, 1.0)
