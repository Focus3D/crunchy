Build a Square


def build_square():
  for i in range(4):
    for i in range(5):
      move()
      build_wall_on_left()
    move()
    turn_left()
  turn_left()
  for i in range(4):
    for i in range(5):
      move()
    move()
    turn_right()
build_square()

Zig Zag




def zig_zag_and_build():
  for i in range(5):
    build_wall_on_left()
    move()
    turn_left()
    move()
    build_wall_on_left()
    turn_right()

def turn_the_corner():
  turn_left()
  move()
  turn_left()
  move()
  turn_left()
  move()

def zag_zig():
  for i in range(4):
    turn_right()
    move()
    turn_left()
    move()

zig_zag_and_build()
turn_the_corner()
zag_zig()
